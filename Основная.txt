<script>
(() => {
  const qs = (s, p = document) => p.querySelector(s);
  const qsa = (s, p = document) => [...p.querySelectorAll(s)];

  // Mobile burger: simply toggles nav visibility by adding a class on body
  const burger = qs("[data-burger]");
  if (burger) {
    burger.addEventListener("click", () => {
      document.body.classList.toggle("is-menu");
      const nav = qs(".nav");
      if (nav) nav.style.display = document.body.classList.contains("is-menu") ? "flex" : "";
    });
  }

  // Reveal on scroll
  const reveals = qsa(".reveal");
  const io = new IntersectionObserver((entries) => {
    entries.forEach(e => {
      if (e.isIntersecting) e.target.classList.add("is-in");
    });
  }, { threshold: 0.12 });
  reveals.forEach(el => io.observe(el));

  // Floating CTA shows after scroll a bit (conversion)
  const floating = qs("[data-floating-cta]");
  if (floating) {
    const onScroll = () => {
      const y = window.scrollY || document.documentElement.scrollTop;
      floating.classList.toggle("is-visible", y > 520);
    };
    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });
  }

  // Modals
  const openModal = (name) => {
    const m = qs(`[data-modal="${name}"]`);
    if (!m) return;
    m.classList.add("is-open");
    m.setAttribute("aria-hidden", "false");
    document.body.style.overflow = "hidden";
  };

  const closeModal = (modalEl) => {
    modalEl.classList.remove("is-open");
    modalEl.setAttribute("aria-hidden", "true");
    document.body.style.overflow = "";
  };

  qsa("[data-open-modal]").forEach(btn => {
    btn.addEventListener("click", () => openModal(btn.getAttribute("data-open-modal")));
  });

  qsa("[data-modal]").forEach(modal => {
    qsa("[data-close-modal]", modal).forEach(el => {
      el.addEventListener("click", () => closeModal(modal));
    });
  });

  window.addEventListener("keydown", (e) => {
    if (e.key !== "Escape") return;
    const open = qs(".modal.is-open");
    if (open) closeModal(open);
  });

  // Catalog filters
  const chips = qsa("[data-filter]");
  const products = qsa(".product");
  if (chips.length && products.length) {
    chips.forEach(ch => {
      ch.addEventListener("click", () => {
        chips.forEach(c => c.classList.remove("is-active"));
        ch.classList.add("is-active");

        const f = ch.getAttribute("data-filter");
        products.forEach(p => {
          const tags = (p.getAttribute("data-tags") || "").split(" ");
          const show = (f === "all") || tags.includes(f);
          p.style.display = show ? "" : "none";
        });
      });
    });
  }

  // Product modal content
  qsa("[data-product]").forEach(btn => {
    btn.addEventListener("click", () => {
      const title = btn.getAttribute("data-title") || "";
      const desc = btn.getAttribute("data-desc") || "";
      const price = btn.getAttribute("data-price") || "";
      const img = btn.getAttribute("data-img") || "";

      const modal = qs('[data-modal="product"]');
      if (!modal) return;

      const t = qs("[data-product-title]", modal);
      const d = qs("[data-product-desc]", modal);
      const p = qs("[data-product-price]", modal);
      const i = qs("[data-product-img]", modal);

      if (t) t.textContent = title;
      if (d) d.textContent = desc;
      if (p) p.textContent = price;
      if (i) { i.src = img; i.alt = title; }

      openModal("product");
    });
  });

  // Fake submit (replace with your backend / telegram bot / email service)
  qsa("[data-form]").forEach(form => {
    form.addEventListener("submit", (e) => {
      e.preventDefault();
      const btn = qs('button[type="submit"]', form);
      if (btn) {
        const old = btn.textContent;
        btn.textContent = "Отправлено ✓";
        btn.disabled = true;
        setTimeout(() => {
          btn.textContent = old;
          btn.disabled = false;
          form.reset();
        }, 1600);
      }
    });
  });
})();
</script>
