    });
  }
  if (nav) {

    });

  // Reveal on scroll

tags.push(typeTag);

  const makeRow = (title, keys, rowKey) => {
    const row = document.createElement("div");
    row.style.display = "flex";
    row.style.gap = "10px";
    row.style.flexWrap = "wrap";
    row.dataset.row = rowKey;

    const head = document.createElement("span");
    head.className = "muted";
    head.style.fontSize = "12px";
    head.style.padding = "10px 2px";
    head.textContent = title;
    row.appendChild(head);

    row.insertAdjacentHTML("beforeend",
      `<button class="chip is-active" type="button" data-filter="${rowKey}:all">Все</button>`
    );

    keys.filter(k => set.has(k)).forEach((k) => {
      row.insertAdjacentHTML("beforeend",
        `<button class="chip" type="button" data-filter="${rowKey}:${k}">${TYPE_LABEL[k] || k}</button>`
      );
    });

    return row;
  };

  // Удаляем всё, что было нарисовано ранее (строки), но оставляем input
  filtersBox.querySelectorAll('[data-row]').forEach(n => n.remove());

  const rowType = makeRow("Тип:", TYPE_ORDER, "type");
  filtersBox.appendChild(rowType);

  // Состояние: только тип + поиск
  const state = { type: "all", q: "" };

  const apply = () => {
    const q = (state.q || "").toLowerCase().trim();

    products.forEach((p) => {
      const btn = qs("[data-product]", p);
      const title = (btn?.getAttribute("data-title") || "").toLowerCase();
      const desc  = (btn?.getAttribute("data-desc") || "").toLowerCase();

      const tags = (p.getAttribute("data-tags") || "").split(/\s+/).filter(Boolean);
      const okType = state.type === "all" || tags.includes(state.type);
      const okQ = !q || (title + " " + desc).includes(q);

      p.style.display = (okType && okQ) ? "" : "none";
    });
  };

  // Кнопки только в строке type
  qsa('.chip[data-filter]', rowType).forEach((ch) => {
    ch.addEventListener("click", () => {
      qsa(".chip", rowType).forEach(b => b.classList.remove("is-active"));
      ch.classList.add("is-active");

      const [, val] = (ch.dataset.filter || "").split(":");
      state.type = val || "all";
      apply();
    });
  });

  // Поиск
  if (searchInput) {
    searchInput.addEventListener("input", () => {
      state.q = searchInput.value;
      apply();
    });
  }

  apply();
}



  // Product modal content
  qsa("[data-product]").forEach(btn => {
    btn.addEventListener("click", () => {
      const title = btn.getAttribute("data-title") || "";
      const desc = btn.getAttribute("data-desc") || "";
      const price = btn.getAttribute("data-price") || "";
      const img = btn.getAttribute("data-img") || "";

      const modal = qs('[data-modal="product"]');
      if (!modal) return;

      const t = qs("[data-product-title]", modal);
      const d = qs("[data-product-desc]", modal);
      const p = qs("[data-product-price]", modal);
      const i = qs("[data-product-img]", modal);

      if (t) t.textContent = title;
      if (d) d.textContent = desc;
      if (p) p.textContent = price;
      if (i) { i.src = img; i.alt = title; }

      openModal("product");
    });
  });

  // Fake submit (replace with your backend / telegram bot / email service)
  qsa("[data-form]").forEach(form => {
    form.addEventListener("submit", (e) => {
      e.preventDefault();
      const btn = qs('button[type="submit"]', form);
      if (btn) {
        const old = btn.textContent;
        btn.textContent = "Отправлено ✓";
        btn.disabled = true;
        setTimeout(() => {
          btn.textContent = old;
          btn.disabled = false;
          form.reset();
        }, 1600);
      }
    });
  });
  
})();
